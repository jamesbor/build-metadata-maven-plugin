27  import java.util.Properties;
28  
29  import org.apache.maven.plugin.MojoExecutionException;
30  import org.apache.maven.plugins.annotations.LifecyclePhase;
31  import org.apache.maven.plugins.annotations.Mojo;
32  import org.apache.maven.plugins.annotations.Parameter;
33  
34  /**
35   * Retrieve current username and place it under a configurable project property
36   * 
37   * @author <a href="kama@soebes.de">Karl-Heinz Marbaise</a>
38   * @since 1.9
39   */
40  @Mojo( name = "username", defaultPhase = LifecyclePhase.VALIDATE, threadSafe = true )
41  public class UserNameMojo
42      extends AbstractDefinePropertyMojo
43  {
44  
45      /**
46       * The name of the property in which to store the local user name.
47       */
48      @Parameter( defaultValue = "build.environment" )
49      private String propertyPrefix;
50  
51      /**
52       * to be called from Maven.
53       */
54      public void execute()
55          throws MojoExecutionException
56      {
57  
58          Properties buildEnvironmentProperties = new Properties();
59  
60          BuildEnvironmentMetaData buildEnvironment =
61              new BuildEnvironmentMetaData( getLog(), getProject(), getSession(), getRuntime(), getDefaultPropertyValue() );
62  
63          buildEnvironment.getUserNameProperty( buildEnvironmentProperties, propertyPrefix );
64  
65          buildEnvironment.defineProjectProperty( buildEnvironmentProperties );
66  
67      }
68  }
